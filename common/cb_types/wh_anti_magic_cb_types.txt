antimagic_revolt = {
	name = CB_antimagic_revolt
	war_name = WAR_NAME_antimagic_revolt
	sprite = 12
	truce_days = 3650
	major_revolt = yes
	allow_whitepeace = no
	
	attacker_can_call_allies = no
	can_ask_to_join_war = no
	
	is_independence = yes
	
	can_use = {
		ROOT = {
			vassal_of = FROM
		}
	}
	on_add = {
		hidden_tooltip = { fire_haruspicy_event_effect = yes }
	}

	is_valid = {
	
		# The defender is not in any black crusades, otherwise the joining effect for defenders will join the wrong war
		FROM = {
			NOT = {
				any_war = {
					using_cb = new_crusade
				}
			}
		}
		ROOT = {
			OR = {
				liege = {
					character = PREV # either independent
				}
				liege = { 
					FROM = { 
						is_liege_or_above = PREV # or have shared liege
					}
				}
			}
		}
		FROM = {
			any_demesne_title = {
				NOT = { lower_tier_than = ROOT }
				temporary = no
			}
			independent = yes
			in_revolt = no
			liege_before_war = {
				in_revolt = no
				liege_before_war = {
					in_revolt = no
				}
			}
		}
	}

	on_success = {
		FROM = {
			set_character_flag = magic_laws_forced
		}
		if = {
			limit = {
				FROM = {
					has_law = magic_law_2
				}
			}
			FROM = {
				if = {
					limit = { has_character_flag = GivingMoreMagicRights }
					primary_title = {
						add_law = {
							law = magic_law_1
							cooldown = no
							opinion_effect = no
						}
					}
					revert_magic_legal_flag = yes
				}
				else = {
					#4th level
				}				
			}
		}
		else_if = {
			limit = {
				FROM = {
					has_law = magic_law_1
				}
			}
			FROM = {
				if = {
					limit = { has_character_flag = GivingMoreMagicRights }
					primary_title = {
						add_law = {
							law = magic_law_0
							cooldown = no
							opinion_effect = no
						}
					}
				}
				else = {
					primary_title = {
						add_law = {
							law = magic_law_2
							cooldown = no
							opinion_effect = no
						}
					}
				}	
			}
		}
		else_if = {
			limit = {
				FROM = {
					has_law = magic_law_0
				}
			}
			FROM = {
				primary_title = {
					add_law = {
						law = magic_law_1
						cooldown = no
						opinion_effect = no
					}
				}
			}
		}
		FROM = {
			clr_character_flag = magic_laws_forced
		}
		
		any_attacker = {
			clr_character_flag = flag_antimagic_revolter
		}

		hidden_tooltip = {
			ROOT = {
				clr_character_flag = flag_antimagic_revolter
				prestige = 100
				piety = 100
				any_demesne_title = {
					add_pressed_claim = FROM
				}
			}
		}
		
		FROM = {
			if = {
				limit = {
					is_invincible_trigger = yes
				}
				antideath_effect = yes
			}
			else = {
				add_trait = staked
				death = { death_reason = death_execution_burning killer = ROOT }
			}
		}
		
		any_attacker = {
			limit = { character = ROOT }
			participation_scaled_prestige = 100
			participation_scaled_piety = 100
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { 
				participation_scaled_prestige = 100
				participation_scaled_piety = 100

			}
		}
	}

	on_reverse_demand = {
		hidden_tooltip = {
			any_attacker = {
				clr_character_flag = flag_antimagic_revolter
			}
			FROM = { 
				narrative_event = { id = wh_magic_law.11 } 
			}
			any_realm_lord = { narrative_event = { id = wh_magic_law.11 } }
			any_player = {
				if = {
					limit = {
						NOT = { character = FROM }
					}
					narrative_event = { id = wh_magic_law.11 } 
				}			
			}
		}
		ROOT = {
			prisoner = FROM
		}
		any_attacker = {
			hidden_tooltip = { 
				participation_scaled_prestige = -100
				participation_scaled_piety = -100 
			}
		}
		FROM = {
			prestige = 100
			piety = 100
			hidden_tooltip = { 
				if = { 
					limit = { NOT = { has_opinion_modifier = { modifier = opinion_traitor who = ROOT } } } #Allow free revocation of a title.
					opinion = {
						who = ROOT
						modifier = opinion_traitor
					}
				}
			}
		}
	}
	on_invalidation = {
		any_attacker = {
			clr_character_flag = flag_antimagic_revolter
		}
	} 

	attacker_ai_victory_worth = {
		factor = -1 #always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 150
	}

	defender_ai_victory_worth = {
		factor = -1 #always accept
	}
	
	defender_ai_defeat_worth = {
		factor = 100
	}
}

impose_magic_law_0 = {
	name = CB_impose_magic_law_0
	war_name = WAR_NAME_impose_magic_law_0
	sprite = 12
	is_permanent = yes
	hostile_against_others = yes
	can_ask_to_join_war = no
	display_on_map = no
	allowed_to_target_tributaries = no

	defender_unoccupied_warscore = yes

	sort_priority = 2000
	
	can_use = {
		ROOT = {
			has_magic_law_3 = no
			independent = yes
			OR = {
				AND = {
					ai = no
					has_magic_law_0 = yes
				}
				AND = {
					trait = witch_hunter
					has_landed_title = k_witch_hunters_order
					magic_legal_50_years = no
				}
			}
		}
		FROM = {
			independent = yes
			in_revolt = no
			has_magic_law_1_or_above = yes
		}
	}
	can_use_gui = {
		ROOT = {
			piety = 1000
		}	
	}
	on_add = {
		hidden_tooltip = { fire_haruspicy_event_effect = yes }
		ROOT = {piety = -1000}
	}

	is_valid = {
		# The defender is not in any black crusades, otherwise the joining effect for defenders will join the wrong war
		FROM = {
			NOT = {
				any_war = {
					using_cb = new_crusade
				}
			}
		}
		ROOT = {
			independent = yes
			has_magic_law_0 = yes
		}
		FROM = {
			independent = yes
			in_revolt = no
			has_magic_law_1_or_above = yes
		}
	}

	on_success = {
		FROM = {
			set_character_flag = magic_laws_forced
		}
		FROM = {
			primary_title = {
				add_law = {
					law = magic_law_0
					cooldown = no
					opinion_effect = no
				}
			}
			if = {
				limit = {
					is_trained_mage_trigger = yes
				}
				if = {
					limit = {
						is_invincible_trigger = yes
					}
					antideath_effect = yes
				}
				else = {
					add_trait = staked
					death = { death_reason = death_execution_burning killer = ROOT }
				}
			}			
		}
		FROM = {
			clr_character_flag = magic_laws_forced
		}

		hidden_tooltip = {
			ROOT = {
				prestige = 100
				piety = 100
				any_demesne_title = {
					add_pressed_claim = FROM
				}
			}
		}
		
		any_attacker = {
			limit = { character = ROOT }
			participation_scaled_prestige = 100
			participation_scaled_piety = 100
		}
		any_attacker = {
			limit = { NOT = { character = ROOT } }
			hidden_tooltip = { 
				participation_scaled_prestige = 100
				participation_scaled_piety = 100

			}
		}
	}

	on_reverse_demand = {
		ROOT = {
			transfer_scaled_wealth = {
				to = FROM
				value = 4.0
			}
		}
		any_attacker = {
			hidden_tooltip = { 
				participation_scaled_prestige = -100
				participation_scaled_piety = -100 
			}
		}
	}
	on_invalidation = {

	}

		factor = -1 #always accept
	}
	
	attacker_ai_defeat_worth = {
		factor = 150
	}

	defender_ai_victory_worth = {
		factor = -1 #always accept
	}

		factor = 100
	}

			}
		}
		modifier = {
