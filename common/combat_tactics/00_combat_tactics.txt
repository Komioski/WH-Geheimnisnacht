harass_tactic = {
	days = 8
	sprite = 4
	group = harass

	trigger = {
		phase = skirmish
		OR = {
			troops = {
				who = light_cavalry
				value = 0.01
			}
			troops = {
				who = camel_cavalry
				value = 0.01
			}
		}
		OR = {
			NOR = {
				troops = {
					who = light_cavalry
					value = 0.45
				}
				troops = {
					who = camel_cavalry
					value = 0.45
				}
			}
			AND = {
				flank_has_leader = yes
