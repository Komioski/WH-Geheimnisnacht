########################################################
# MINOR TITLES
# Characters can have any number of these
########################################################
#	grant_limit: Max number of holders.
#	realm_in_name: Append the name of the country after the title (Queen [of Sweden])
#	allowed_to_hold: conditions for the character to hold the title
#	allowed_to_grant: conditions for a ruler to grant the title
#	regent: This is the title for regents (not really "minor")
#	gain_effect: fires when the title is granted
#	lose_effect: fires when a title is lost through most causes
#	retire_effect: fires when a title is lost due to the character stepping down
#	death_effect: fires when a title is lost due to the character dying
#	message: should fire a message to the its owner when set
#	The other fields are all character modifiers.

# Commander, granted via its own interaction, so very special
title_commander = {
	is_high_prio = yes

	attribute = martial
	dignity = 0.05
	opinion_effect = 5
	realm_in_name = yes

	monthly_salary = 0.05
	monthly_prestige = 0.50

	revoke_allowed = no # Handled by diplomatic interaction

	allowed_to_hold = {
		liege = { character = FROM }
		is_adult = yes
		prisoner = no
		is_incapable = no
		is_inaccessible_trigger = no
		NOT = {
			primary_title = {
				temporary = yes
			}
		}
		#Only some factions allow female commanders
		trigger_if = {
			limit = { is_female = yes }
			OR = {
				AND = {
					trait = shieldmaiden
					liege = { religion = ROOT }
				}
				AND={
					trait = creature_saurus
					liege = { religion_group = old_ones_group }
				}
				liege = {
					OR = {
						religion = amazon
						religion_group = cytharai_group
						religion_group = asrai_cadai
						primary_title = {
							OR = {
								has_law = status_of_women_4
								has_law = enatic_succession
								has_law = enatic_cognatic_succession
								has_law = true_cognatic_succession
							}
						}
					}
				}
				
			}
		}
		#CONDITIONAL: Dawi age requirement
		conditional_tooltip = {
			trigger = {
				OR = {
					trait = creature_dwarf
					trait = creature_chaos_dwarf
					liege = {
						OR = {
							trait = creature_dwarf
							trait = creature_chaos_dwarf
						}
					}
				}
			}
			custom_tooltip = {
				text = dwarf_commander_age_tt
				OR = {
					NOR = {
						trait = creature_dwarf
						trait = creature_chaos_dwarf
					}
					age = 75
					trait = chaos_dwarf_bull_centaur
				}
			}
		}
		#CONDITIONAL: Amazon female only requirement
		conditional_tooltip = {
			trigger = {
				OR = {
					trait = creature_amazon
					liege = { trait = creature_amazon }
				}
			}
			is_female = yes
		}
		#CONDITIONAL: Beastmen only requirement
		conditional_tooltip = {
			trigger = {
				OR = {
					trait = creature_beastman
					liege = { trait = creature_beastman }
				}
			}
			trait = creature_beastman
		}
		#CONDITIONAL: Orcs and Gobbos only requirement
		conditional_tooltip = {
			trigger = {
				OR = {
					culture_group = goblin_group
					culture_group = orc_group
					liege = {
						OR = {
							culture_group = goblin_group
							culture_group = orc_group
						}
					}
				}
			}
			OR = {
				culture_group = orc_group
				culture_group = goblin_group
			}
		}
		#CONDITIONAL: Lizardmen only requirement
		conditional_tooltip = {
			trigger = {
				OR = {
					trait = creature_slann
					trait = creature_skink
					trait = creature_saurus
					liege = {
						OR = {
							trait = creature_slann
							trait = creature_skink
							trait = creature_saurus
						}
					}
				}
			}
		}
		OR = {
			has_character_flag = special_marshal
			# Ogres get special treatment
			culture_group = ogre_group
			# Change to something more loose
			religion_group = FROM
			# Mages can be commanders
			custom_tooltip = {
				text = commander_with_magic_tt
				OR = {
					has_character_modifier = magic_power_1
					has_character_modifier = magic_power_2
					has_character_modifier = magic_power_3
					has_character_modifier = magic_power_4
				}
			}
    	}
	}

	gain_effect = {
		if = {
			limit = {
				liege = {
					OR={
					has_law = succ_military_elective
					has_law = succ_byzantine_elective
					}
				}
			}
			character_event = { id = HFP.20805 days = 1 }
		}
		if={
			limit ={
				trait = creature_elf
			}
			hidden_effect = { # Because this cannot be added in the passive effects, for some code reason
				add_character_modifier = {
					name = hidden_commander_elf
					duration = -1
					hidden = yes
				}
			}
		}
		else_if={
			limit ={
				OR={
					trait = creature_chaos_dwarf
					trait = creature_dwarf
				}
			}
			hidden_effect = { # Because this cannot be added in the passive effects, for some code reason
				add_character_modifier = {
					name = hidden_commander_dawi
					duration = -1
					hidden = yes
				}
			}
		}
		else_if={
			limit ={
				culture_group = imperial_group
				NOT={culture=neueslander}
				trait = creature_human
				religion_group = old_world_group
			}
			hidden_effect = { # Because this cannot be added in the passive effects, for some code reason
				add_character_modifier = {
					name = hidden_commander_bonus_empire
					duration = -1
					hidden = yes
				}
			}
		}
		else_if={
			limit ={
				culture_group = estalian_group
				NOT={culture=comarcano}
				trait = creature_human
				is_undead_trigger = no
			}
			hidden_effect = { # Because this cannot be added in the passive effects, for some code reason
				add_character_modifier = {
					name = hidden_commander_bonus_estalian
					duration = -1
					hidden = yes
				}
			}
		}
		else_if={
			limit ={
				culture_group = tilean_group
				NOT={culture=oltremari}
				trait = creature_human
				religion_group = old_world_group
			}
			hidden_effect = { # Because this cannot be added in the passive effects, for some code reason
				add_character_modifier = {
					name = hidden_commander_bonus_tilean
					duration = -1
					hidden = yes
				}
			}
		}
		else = {
			hidden_effect = { # Because this cannot be added in the passive effects, for some code reason
				add_character_modifier = {
					name = hidden_commander_bonus
					duration = -1
					hidden = yes
				}
			}
		}
		character_event = { id = asur.53 }
	}
	lose_effect = {
		if = {
			limit = {
				liege = {
					OR={
					has_law = succ_military_elective
					has_law = succ_byzantine_elective
					}
				}
			}
			character_event = { id = HFP.20803 days = 1 } #Make sure the commander is not being made a sub-vassal.

		}
		character_event = { id = asur.531 }
		hidden_effect = {
			remove_character_modifier = hidden_commander_bonus
			remove_character_modifier = hidden_commander_bonus_tilean
			remove_character_modifier = hidden_commander_bonus_estalian
			remove_character_modifier = hidden_commander_bonus_empire
			remove_character_modifier = hidden_commander_elf
			remove_character_modifier = hidden_commander_dawi
		}
	}

	message = yes
}
