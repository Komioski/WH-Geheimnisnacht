		}

			factor = 0.25
			not = { martial = 5 }
		}
		modifier = {
			factor = 0
			not = { martial = 3 }
		}
		modifier = {
			factor = 1.25
			martial = 10
		}
		modifier = {
			factor = 1.25
			martial = 13
		}
		modifier = {
			factor = 1.25
			martial = 16
		}
		modifier = {
			factor = 1.25
			martial = 19
		}
		modifier = {
			factor = 1.25
			martial = 22
		}
	}
	success = {
		custom_tooltip = {
			text = DR_obj_win_war
			hidden_tooltip = { has_character_flag = obj_win_war_won_war }
		}
	}
	abort = {
		or = {
			is_ruler = no
			prisoner = yes
			trait = incapable
		}
	}
	abort_effect = {
	}
	effect = {
		if = {
			limit = { not = { has_character_flag = obj_win_war } }
			change_martial = 1
			prestige = 50
		}
		if = {
			limit = { has_character_flag = obj_win_war }
			prestige = 25
		}
		set_character_flag = obj_win_war
		clr_character_flag = obj_win_war_won_war
	}
}

obj_norscan_tribe_become_stronghold = {
	type = character
	can_cancel = yes

	potential = {
		government=norscan_government
		NOT={has_character_flag=norscan_overlord_completion}
		OR={
			independent = yes
			ai=no
		}
		prestige=100
	}

	allow = {
		NOT = { is_inaccessible_trigger = yes }
	}

	chance = {
		factor = 200

		modifier = {
			factor = 0
			trait = cynical
		}

		modifier = {
			factor = 0.5
			NOT = {
				trait = zealous
			}
		}

	}

	success = {
		higher_tier_than=COUNT
		prestige=2000
		OR={
			AND={
				num_of_duke_titles >= 3
				prestige=5000
			}
			custom_tooltip = {
				text = own_10_provinces_in_norsca
				any_realm_province={
					culture_group=norscan_group
					location = { region = world_norsca }
					count=10
				}
			}
			custom_tooltip = {
				text = own_5_neighbour_tributaries_norsca
				any_neighbor_independent_ruler = {
					count >= 5
					culture_group=norscan_group
					is_tributary = {
						type = default
						suzerain = FROM
					}
				}
			}
			custom_tooltip = {
				text = own_10_tributaries_norsca
				any_independent_ruler = {
					count >= 10
					culture_group=norscan_group
					is_tributary = {
						type = default
						suzerain = FROM
					}
				}
			}
		}
	}

	abort = {
		OR={
			trait = incapable
			trait = content
			NOT={prestige=100}
		}
	}

	abort_effect = {
		prestige = -500
	}

	effect = {
		if={
			limit={
				government=norscan_government
			}
			set_government_type=norscan_government_feudal
		}
		if={
			limit={
				government=norscan_government
				capital_scope={has_castle=no}
			}
			
			capital_scope={convert_to = CASTLE}
		}
		set_character_flag=norscan_overlord_completion
		prestige = 500
	}
}
