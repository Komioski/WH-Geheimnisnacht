check_if_mage_should_be_burned = {
	#For characters abroad
	NOR = {
		AND = {
			is_trained_mage = yes
			OR = {
				liege = { has_magic_law_2 = yes }
				dynasty_head = { has_magic_law_2 = yes }
				AND = {
					is_ruler = no
					OR = {
						mother_even_if_dead = { is_ruler = yes has_magic_law_2 = yes }
						mother_even_if_dead = { OR = { mother_even_if_dead = { is_ruler = yes has_magic_law_2 = yes } father_even_if_dead = { is_ruler = yes has_magic_law_2 = yes } } }
						father_even_if_dead = { is_ruler = yes has_magic_law_2 = yes }
						father_even_if_dead = { OR = { father_even_if_dead = { is_ruler = yes has_magic_law_2 = yes } mother_even_if_dead = { is_ruler = yes has_magic_law_2 = yes } } }
					}
				}
			}
		}
		AND = {
			is_trained_mage = no
			has_magic_potential = yes
			NOT = { has_character_modifier = magic_potential_successfully_hidden }
			OR = {
				liege = { has_magic_law_1 = yes }
				dynasty_head = { has_magic_law_1 = yes }
				AND = {
					is_ruler = no
					OR = {
						mother_even_if_dead = { is_ruler = yes has_magic_law_1 = yes }
						mother_even_if_dead = { OR = { mother_even_if_dead = { is_ruler = yes has_magic_law_1 = yes } father_even_if_dead = { is_ruler = yes has_magic_law_1 = yes } } }

						father_even_if_dead = { OR = { father_even_if_dead = { is_ruler = yes has_magic_law_1 = yes } mother_even_if_dead = { is_ruler = yes has_magic_law_1 = yes } } }

	}
	NOT = {  top_liege = { has_character_modifier = expelled_k_witch_hunters_order } }

	NOT = { top_liege = {  trait = peasant_leader } }

				NOT = { trait = BretonDamsel }
				gender_is_allowed_magic = no
				NAND = {
					has_global_flag = magic_legal_old_world_50_years
					liege = {
						OR = {
							mercenary = yes
							is_theocracy = yes
							government = congregation_government
						}
					}
				}
			}
			AND = {
				is_trained_mage = yes
				gender_is_allowed_magic = no
				#The Ice witchs are tolerated in the Empire to avoid pissing off a useful ally
				NOT = { trait = ice_witch }
				NOT = { trait = BretonDamsel }
				OR = {
					has_magic_law_0 = yes

				}

gender_is_allowed_magic = {
	custom_tooltip = {
		text = gender_allowed_magic_tt
		OR = {
			AND = {
				is_female = yes
				OR = {
					AND = {
						human_religion_that_allows_female_mages = yes
						liege = {
							human_religion_that_allows_female_mages = yes
						}
					}
					top_liege = { trait = malekith }
				}
			}
			AND = {
				is_female = no
				AND = {
					human_religion_that_allows_male_mages = yes
					liege = {
						human_religion_that_allows_male_mages = yes
					}
				}
				NOT = { top_liege = { trait = malekith } }
			}
			has_magic_law_any = no
			has_magic_law_2 = yes
		}
	}
}

supports_legal_magic = {
	OR = {
		hidden_tooltip = {
			is_human = no
		}
		is_trained_mage_or_has_magic_potential = yes
		#Made too easy to pass laws at start
		#hidden_tooltip = {
		#	secret_religion_group = chaos_gods_group
		#}
		AND = {
			#Realms with law 0 are more inclined to hate magic.
			has_magic_law_0 = yes
			hates_legal_magic = no
		}
		calc_true_if = {
			amount = 2
			custom_tooltip = {
				text = bloodline_effect_magic_legacy
				any_owned_bloodline = {
					has_bloodline_flag = magic_legacy
					bloodline_is_active_for = PREV
				}
			}
			trait = kind
			trait = just
			trait = scholar
			trait = mystic
			is_doom_high_trigger = yes
			is_doom_very_high_trigger = yes
			any_spouse = {
				is_trained_mage_or_has_magic_potential = yes
			}
			father_even_if_dead = {
				is_trained_mage_or_has_magic_potential = yes
			}
			mother_even_if_dead = {
				is_trained_mage_or_has_magic_potential = yes
			}
			any_child  = {
				is_trained_mage_or_has_magic_potential = yes
			}
			any_sibling  = {
				is_trained_mage_or_has_magic_potential = yes
			}
			any_friend = {
				OR = {
					is_trained_mage_or_has_magic_potential = yes
					trait = creature_elf
				}
			}
			any_lover = {
				OR = {
					is_trained_mage_or_has_magic_potential = yes
					trait = creature_elf
				}
			}
			has_global_flag = magic_legal_old_world_50_years
		}
	}
}

hates_legal_magic = {
	NOT = { is_trained_mage_or_has_magic_potential = yes }
	hidden_tooltip = {
		is_human = yes
	}
	OR = {
		trait = zealous
		OR = {
			AND = {
				has_magic_law_0 = yes
				NOT = { learning = 16 }
			}
			AND = {
				has_magic_law_1_or_above = yes
				NOT = { learning = 12 }
			}
		}
	}
	OR = {
		AND = {
			has_magic_law_0 = yes
			NOT = {
				has_global_flag = magic_legal_old_world_50_years
			}
			OR = {
				trait = wroth
				trait = cruel
				trait = impaler
				trait = arbitrary
				trait = paranoid
				trait = detached_priest
				trait = martial_cleric
				trait = trusting
				trait = shy
				trait = craven
				trait = envious
				ai_rationality < 0
				is_doom_very_low_trigger = yes
				is_doom_low_trigger = yes
				AND = {
					is_theocracy = yes
					NOT = { trait = kind }
				}
			}
		}
		AND = {
			OR = {
				has_magic_law_1 = yes
				AND = {
					has_magic_law_0 = yes
					has_global_flag = magic_legal_old_world_50_years
				}
			}
			OR = {
				trait = wroth
				trait = cruel
				trait = impaler
				trait = arbitrary
				trait = paranoid
				trait = detached_priest
				trait = martial_cleric
				ai_rationality < 0
				is_doom_very_low_trigger = yes
				is_doom_low_trigger = yes
				AND = {
					is_theocracy = yes
					NOT = { trait = kind }
				}
			}
		}
		AND = {
			has_magic_law_2 = yes
			calc_true_if = {
				amount = 2
				trait = wroth
				trait = cruel
				trait = impaler
				trait = arbitrary
				trait = paranoid
				trait = detached_priest
				trait = martial_cleric
				is_dumb_trigger = yes
				ai_rationality < 0
				AND = {
					is_theocracy = yes
					is_dumb_trigger = yes
				}
			}
		}
	}
}
