can_use_retinue_trigger = {
	OR = {
		ai = no
		is_patrician = yes
		OR={
			AND={
				independent=yes
				prestige=500
				OR={
					government=beastman_government
					government=chaos_horde_government
				}
				is_nomadic = yes
			}
			AND={
				independent=no
				prestige=250
				is_nomadic = yes
			}
			total_years_played <= 1
		}
		AND = {
			higher_real_tier_than = BARON
			OR = {
				AND = {
					OR = {
						is_feudal = yes
						is_theocracy = yes
						is_republic = yes
					}
					# > doesn't work with demesne_size
					OR = {
						demesne_size = 2
						higher_real_tier_than = COUNT
					}
				}
				is_tribal = yes
			}
		}
	}
}

			any_artifact={
				artifact = heart_of_avelorn_everqueen
				is_artifact_active = yes
			}
			NOT={ has_character_flag = exhausted_heart }
		}
		AND={
			ai = yes
			OR = {
				trait = chosen_khorne
				trait = chosen_nurgle
				trait = chosen_slaanesh
				trait = chosen_tzeentch
				is_demon_trigger = yes
			}
			NOT={ 
				check_variable = { 
					which = cheated_death value > 1 
				}
			}
		}		
	}
