negative_money = {
	potential = {
		is_ruler = yes
	}

	trigger = {
		OR = {
			NOT = { wealth = 0 }
			AND = {
				trait = creature_dwarf
				NOT = { wealth = 10 }
			}
		}
	}

	icon = 24

	land_morale = -0.25

	active = no
}
