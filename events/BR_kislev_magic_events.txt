character_event = {
	#Basic event information
	id = wardrobe.0
	desc = EVTDESCwardrobe.0
	border = GFX_event_normal_frame_intrigue

	#Fast event triggers
	only_playable = yes
	max_age = 14
	min_age = 6
	prisoner = no

	trigger = {
		AND = {
			culture_group = kislevite_group
			is_female = yes
			NOT = { has_character_flag = discovered_wardrobe }
		}
	}
	mean_time_to_happen = {
		years = 50
		modifier = {
		factor = 0.8
		has_trait = curious
		}
		modifier = {
		factor = 0.5
		intrigue = 8
		}
	}
	option = {
		name = EVTOPTwardrobe.0
		character_event = { id = wardrobe.1 }
	}
}
