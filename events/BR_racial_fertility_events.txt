namespace = health

#Both following events are activated on_post_birth
#elven infertility - each time they get pregnant, there is a chance it is their last child they can have#
character_event = {
	id = health.1
	desc = EVTDESCHEALTH.1				# You can feel that this will be your last child.
	picture = "GFX_evt_pregnancy"

	is_triggered_only = yes
	trigger = {
		trait = creature_elf
	}

	immediate = {
		remove_trait = hard_pregnancy
		
	}

	option = {
		name = "OK"		
		if = {
			limit = { 
				OR = {
					culture_group = asur_group 
					culture_group = asrai_group
					culture_group = druchi_group
				}
			}
			random_list = {
				10 = {
					add_trait = infertile
					modifier = {
						factor = 0.8
						culture_group = asrai_group
					}
					modifier = {
						factor = 0.6
						culture_group = druchi_group
					}
					modifier = {
						factor = 0
						num_of_children <= 1
					}
					modifier = {
						factor = 3
						num_of_children >= 2
						num_of_children < 3
					}
					modifier = {
						factor = 6
						num_of_children >= 4
						num_of_children < 5
					}
					modifier = {
						factor = 8
						num_of_children >= 6
					}
					modifier = {
						factor = 1.2
						fertility <= 0.40
					}
					modifier = {
						factor = 0.95
						fertility >= 0.50
						fertility < 0.60
					}
					modifier = {
						factor = 0.90
						fertility >= 0.50
						fertility < 0.60
					}
					modifier = {
						factor = 0.85
						fertility >= 0.60
						fertility < 0.70
					}
					modifier = {
						factor = 0.80
						fertility >= 0.70
					}
				}
				90 = {
					fertility = -0.10
				}
			}
		}
	}
}

##fires for the kid, pings to the mother
character_event = {
	id = health.3
	hide_window = yes

	is_triggered_only = yes
	trigger = {
		OR = { 
			mother = { 
				OR = {
					trait = creature_elf
					trait = creature_dwarf
					trait = creature_chaos_dwarf
				}
				NOT = { trait = infertile }
			}
			father = {
				age = 200
				trait = creature_elf
				NOT = { trait = infertile }
			}
		}
	}

	immediate = {
		if = { 
			limit = { 
				mother = { trait = creature_elf } 
			}
			mother = { character_event = { id = health.1 days = 2 } } 
		}
		else_if = { 
			limit = { father = { trait = creature_elf age = 200 } }
			father = { character_event = { id = health.4 days = 2 } } 
		}
		else_if = { 
			limit = { father = { OR = { trait = creature_dwarf trait = creature_chaos_dwarf } age = 120 } }
			father = { character_event = { id = health.5 days = 2 } } 
		}
		else_if = { 
			limit = { mother = { OR = { trait = creature_dwarf trait = creature_chaos_dwarf } } }
			mother = { character_event = { id = health.2 days = 2 } } 
		}
		if = {
			limit = {
				mother = {
					OR = { 
						trait = creature_dwarf 
						trait = creature_chaos_dwarf
						trait = creature_elf
					}
				}
			}
			mother = {
				if = {
					limit = {
						tier = BARON
					}
					add_character_modifier = {
						modifier = elder_race_fertility_mod_1
						duration = 3650
					}
				}
				else = { 
					add_character_modifier = {
						modifier = elder_race_fertility_mod_2
						duration = 3650
					}
				}
			}
		}
		if = {
			limit = {
				father = {
					OR = { 
						trait = creature_dwarf 
						trait = creature_chaos_dwarf
						trait = creature_elf
					}
				}
			}
			father = {
				if = {
					limit = {
						tier = BARON
					}
					add_character_modifier = {
						modifier = elder_race_fertility_mod_1
						duration = 3650
					}
				}
				else = { 
					add_character_modifier = {
						modifier = elder_race_fertility_mod_2
						duration = 3650
					}
				}
			}
		}
	}
}
