character_event = {
	id = CC.10020
	hide_window = yes

	is_triggered_only = yes

	immediate = { FROM = { society_quest_event = { id = CC.10021 days = 7 random = 7 } } }
}

#Assign event
society_quest_event = {
	id = CC.10021
	desc = {
		trigger = {
			ROOT = {
				has_character_flag = flag_renounce_focus_for_family
			}
		}
		text = EVTDESCACC10021
	}
	desc = {
		trigger = {
			ROOT = {
				has_character_flag = flag_renounce_focus_for_business
			}
		}
		text = EVTDESCBCC10021
	}

	is_triggered_only = yes
	is_in_society = yes
	only_capable = yes

	trigger = {
		society_member_of = cult_of_nurgle
		same_society_as = FROM
		has_any_quest = no
		NOT = { has_character_modifier = quest_cooldown_timer }
		NOT = { is_inaccessible_trigger = yes }
		is_landed = yes
		NOR = {
			has_focus = focus_family
			has_focus = focus_business
		}
	}

	immediate = {
		random_list = {
			50 = { set_character_flag = flag_renounce_focus_for_family }
			50 = { set_character_flag = flag_renounce_focus_for_business }
		}
	}

	quest_target = ROOT

	option = {
		name = ACCEPT
		ROOT = {
			show_scope_change = no
			if = {
				limit = { has_character_flag = flag_renounce_focus_for_family }
				set_quest_target = {
					id = quest_chaos_cult_weak_focus_to_family
					holder = ROOT
				}
			}
			if = {
				limit = { has_character_flag = flag_renounce_focus_for_business }
				set_quest_target = {
					id = quest_chaos_cult_weak_focus_to_business
					holder = ROOT
				}
			}
		}
	}
	option  = {
		name = DECLINE
		custom_tooltip = { text = decline_quest_tooltip }

		add_character_modifier = {
			name = quest_cooldown_timer
			hidden = yes
			days = 1000
		}

		ai_chance = { factor = 0 }
	}
	after = {
		hidden_effect = { #For the one who instantly selected the focus, bypassing the mission-trigger
			trigger_switch = {
				on_trigger = has_character_flag
				flag_renounce_focus_for_family = {
					if = {
						limit = { has_focus = focus_family }
						character_event = { id = CC.10022 days = 1 }
					}
				}
				flag_renounce_focus_for_business = {
					if = {
						limit = { has_focus = focus_business }
						character_event = { id = CC.10023 days = 1 }
					}
				}
			}
		}

		clr_character_flag = flag_renounce_focus_for_family
		clr_character_flag = flag_renounce_focus_for_business
	}
}
