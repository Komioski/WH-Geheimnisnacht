# on_birth event to give ormazdic sayyid/mirza traits
character_event = {
	id = ormazd.0
	desc = ormazdian0desc
	picture = GFX_evt_bishop

	religion_group = ormazd_gods_group

	is_triggered_only = yes

	trigger = {
		NOR = {
			trait = ormazd_sayyid
			trait = non_ormazd_sayyid
			trait = ormazd_mirza
			trait = non_ormazd_mirza
		}
		OR = {
			father_even_if_dead = { trait = ormazd_sayyid }
			mother_even_if_dead = { trait = ormazd_sayyid }
			father_even_if_dead = { trait = non_ormazd_sayyid }
			mother_even_if_dead = { trait = non_ormazd_sayyid }
		}
	}

	option = {
		name = OK
		if = {
			limit = {
				NOT = { religion = ormazd }
				OR = {
					father_even_if_dead = { trait = ormazd_sayyid }
					father_even_if_dead = { trait = non_ormazd_sayyid }
				}
			}
			add_trait = non_ormazd_sayyid
			break = yes
		}
		if = {
			limit = {
				NOT = { religion = ormazd }
				OR = {
					mother_even_if_dead = { trait = ormazd_sayyid }
					mother_even_if_dead = { trait = non_ormazd_sayyid }
				}
			}
			add_trait = non_ormazd_mirza
			break = yes
		}
		if = {
			limit = {
				religion = ormazd
				OR = {
					father_even_if_dead = { trait = non_ormazd_sayyid }
					father_even_if_dead = { trait = ormazd_sayyid }
				}
			}
			add_trait = ormazd_sayyid
			break = yes
		}
		if = {
			limit = {
				religion = ormazd
				OR = {
					mother_even_if_dead = { trait = ormazd_sayyid }
					mother_even_if_dead = { trait = non_ormazd_sayyid }
				}
			}
			add_trait = ormazd_sayyid
			break = yes
		}
	}
}

# on_character_convert_religion maintenance event swapping sayyid/mirza
character_event = {
	id = ormazd.1
	desc = ormazdian1desc
	picture = GFX_evt_bishop

	religion_group = ormazd_gods_group

	is_triggered_only = yes

	trigger = {
		OR = {
			trait = non_ormazd_sayyid
			trait = ormazd_sayyid
			trait = non_ormazd_mirza
			trait = ormazd_mirza
		}
	}

	option = {
		name = OK
		if = {
			limit = {
				religion = ormazd
			}
			if = {
				limit = {
					trait = non_ormazd_sayyid
				}
				add_trait = ormazd_sayyid
				remove_trait = non_ormazd_sayyid
			}
			if = {
				limit = {
					trait = non_ormazd_mirza
				}
				add_trait = ormazd_mirza
				remove_trait = non_ormazd_mirza
			}
		}
		if = {
			limit = {
				NOT = { religion = ormazd }
			}
			if = {
				limit = {
					trait = non_ormazd_sayyid
				}
				remove_trait = ormazd_sayyid
				add_trait = non_ormazd_sayyid
			}
			if = {
				limit = {
					trait = non_ormazd_mirza
				}
				remove_trait = ormazd_mirza
				add_trait = non_ormazd_mirza
			}
		}
	}
}
