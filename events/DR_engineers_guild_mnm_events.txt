			}
			75 = {
				#flimsy
				modifier = {
					factor = 1.2
					is_weak_trigger = yes
				}
				#balance, see above
				modifier = {
					factor = 0.5
					has_injury_trigger = yes
				}
				add_trait = wounded
			}
			10 = {
				#tough people
				modifier = {
					factor = 1.5
					trait = creature_dwarf
				}
				#tough person
				modifier = {
					factor = 1.5

				}

					factor = 2

					limit = {
