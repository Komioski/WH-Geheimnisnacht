# The Kislevite Melting Pot
character_event = {
	id = 55002
	desc = "U_EVTDESC55002"
	picture = GFX_evt_kislev_town

	only_playable = yes

	trigger = {
		should_evolve_cultures_trigger = yes
		OR={
			culture=gospodar
			culture=kislevite
		}
		OR={
			AND={
				any_demesne_province = {
					region = world_old_world_kislev
					culture = gospodar
					any_province_holding = {
						NOT = { holding_type = nomad }
					}
				}
				202 ={# c_kislev
					culture=kislevite
				}
			}
			202 ={# c_kislev
				owner = { character = ROOT }
				NOT={culture=kislevite}
			}
		}
	}

	mean_time_to_happen = {
		months = 1500
		modifier = {
			factor = 0.75
			learning = 7
		}
		modifier = {
			factor = 0.1
			202 ={# c_kislev
				owner = { character = ROOT }
				NOT={culture=kislevite}
			}
		}
		modifier = {
			factor = 0.5
			202 ={# c_kislev
				culture=kislevite
				owner = { character = ROOT }
			}
		}
		modifier = {
			factor = 1.5
			202 ={# c_kislev
				culture=kislevite
				owner = { NOT={character = ROOT }}
			}
		}
		modifier = {
			factor = 5
			202 ={# c_kislev
				NOT={culture=kislevite}
				owner = { NOT={character = ROOT }}
			}
		}
		modifier = {
			factor = 0.75
			learning = 9
		}
		modifier = {
			factor = 0.75
			learning = 11
		}
		modifier = {
			factor = 0.75
			learning = 13
		}
		modifier = {
			factor = 1.5
			NOT = { learning = 5 }
		}
		modifier = {
			factor = 1.5
			NOT = { learning = 3 }
		}
		modifier = {
			factor = 2.0
			NOT = { learning = 1 }
		}
		modifier = {
			factor = 0.3
			OR = {
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots
				}
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots_and_slower_normal
				}
			}
		}
	}

	option = {
		name = "U_EVTOPTA55002"
		if = {
			limit = {
				NOT = { culture = kislevite }
			}
			culture = kislevite
		}
		if={limit={
				any_demesne_title={ title = c_kislev }
				202 ={# c_kislev
					NOT={culture=kislevite}
				}
			}
			202 ={# c_kislev
				culture=kislevite
				religion=root
			}
		}
		else_if={
			limit={
				any_demesne_title={ title = c_kislev }
				202 ={# c_kislev
					random_neighbor_province={
						not={culture=kislevite}
						culture_group=kislevite_group
					}
				}
			}
			202 ={# c_kislev
				random_neighbor_province={
					limit={
						not={culture=kislevite}
						culture_group=kislevite_group
					}
					culture=kislevite
				}
			}
			set_character_flag=works
		}
		else={
			random_demesne_province = {
				limit = {
					region = world_old_world_kislev
					culture = gospodar
					any_province_holding = {
						NOT = { holding_type = nomad }
					}
				}
				culture = kislevite
			}
		}
	}
}

# A ruler is assimilated by the local culture
character_event = {
	id = 55010
	desc = "EVTDESC55010"
	picture = GFX_evt_stone_church

	only_rulers = yes
	min_age = 16
	capable_only = yes
	prisoner = no
	lacks_dlc = "Rajas of India" # Controlled by Decision: convert_to_local_culture

	trigger = {
		NOR = {
			government = nomadic_government
			government = chaos_horde_government
		}
		primary_title = {
			NOT = { conquest_culture = ROOT }
			NOT = { culture = ROOT }
			any_previous_holder = {
				NOT = { character = ROOT } # The first holder of a title does not convert
			}
		}
		capital_scope = {
			NOT = { culture = ROOT }
		}
		top_liege = {
			OR = {
				character = ROOT
				ROOT = {
					capital_scope = {
						culture = PREVPREV
					}
				}
			}
		}
		NOT = { government = celestial_empire_government }
	}

	mean_time_to_happen = {
		months = 1200
		modifier = {
			factor = 0.5
			top_liege = {
				NOT = { character = ROOT }
				ROOT = {
					capital_scope = {
						culture = PREVPREV
					}
				}
			}
		}
		modifier = {
			factor = 0.5
			primary_title = {
				ROOT = {
					capital_scope = {
						culture = PREVPREV
					}
				}
			}
		}
		modifier = {
			factor = 2.0
			capital_scope = {
				culture_group = ROOT # Same culture group: slower assimilation
			}
		}
	}

	option = {
		name = "EVTOPTA55010"

		# Piss off the appropriate vassals
		any_vassal = {
			limit = {
				culture = ROOT
				age = 16
			}
			opinion = {
				who = ROOT
				modifier = opinion_adopted_foreign_ways
				years = 10
			}
			hidden_tooltip = { character_event = { id = 55011 days = 0 } }
		}

		capital_scope = {
			reverse_culture = ROOT
		}
	}

	option = {
		name = "EVTOPTB55010"
		ai_chance = {
			factor = 0
		}
	}
}
