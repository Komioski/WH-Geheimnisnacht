# New Chinese Emperor. Pick a new Policy and send a News Event!
# Hidden. Fired from on_offmap_ruler_changed.
character_event = {
	id = JD.50000
	offmap = only

	has_dlc = "Jade Dragon"

	is_triggered_only = yes
	hide_window = yes

	trigger = { is_offmap_tag = offmap_cathay }

	immediate = {
		FROMFROM = {
			set_offmap_flag = no_policy_news # Don't fire the News event in 'on_offmap_policy_changed'
			random_list = {
				100 = { # OPEN
					modifier = {
						factor = 1.5
						has_status = cathay_stable
					}
					modifier = {
						factor = 5
						has_policy = cathay_open
						NOT = { days_since_policy_change = 10950 } #30 years
					}
					set_policy = cathay_open
					log = "-------------------------------------"
					log = "Cathay Policy Logging:"
					log = "Cathays new Policy is now OPEN"
					log = "-------------------------------------"
				}
				20 = { # ISOLATIONIST
					trigger = {
						OR = {
							NOT = { has_offmap_tmp_flag = cathay_forced_open }
							had_offmap_tmp_flag = { flag = cathay_forced_open years = 50 }
						}
					}
					modifier = {
						factor = 10
						has_offmap_flag = cathay_isolationist_trend
					}
					modifier = { # Hard times... (negative Status)
						factor = 2
						cathay_is_suffering_trigger = yes
					}
					modifier = {
						factor = 0.5
						has_status = cathay_golden_age
					}
					modifier = {
						factor = 5
						has_policy = cathay_isolationist
						NOT = { days_since_policy_change = 10950 } #30 years
					}
					set_policy = cathay_isolationist
					log = "-------------------------------------"
					log = "Cathay Policy Logging:"
					log = "Cathays new Policy is now ISOLATIONIST"
					log = "-------------------------------------"
					clr_offmap_flag = cathay_isolationist_trend
					clr_offmap_tmp_flag = cathay_forced_open
				}
				10 = { # EXPANSIONIST
					trigger = {
						NOT = { has_offmap_flag = cathay_badly_beaten_3 }
						NOT = { check_variable = { which = "global_cathay_change_to_expansionist_count" value = 6 } }
					}
					modifier = {
						factor = 20
						has_offmap_flag = cathay_expansionist_trend
					}
					modifier = {
						factor = 2
						has_status = cathay_golden_age
					}
					modifier = {
						factor = 0.75
						has_offmap_flag = cathay_badly_beaten_1
					}
					modifier = {
						factor = 0.5
						has_offmap_flag = cathay_badly_beaten_2
					}
					modifier = {
						factor = 10
						has_policy = cathay_expansionist
						NOT = { days_since_policy_change = 14600 } #40 years
					}
					modifier = {
						factor = 0.1
						has_policy = cathay_expansionist
						days_since_policy_change = 32850 #90 years
					}
					modifier = {
						factor = 0.2
						check_variable = { which = "global_cathay_change_to_expansionist_count" value = 3 }
					}
					if = {
						limit = { NOT = { has_policy = cathay_expansionist } }
						change_variable = { which = "global_cathay_change_to_expansionist_count" value = 1 }
					}
					set_policy = cathay_expansionist
					log = "-------------------------------------"
					log = "Cathay Policy Logging:"
					log = "Cathays new Policy is now EXPANSIONIST"
					log = "-------------------------------------"
					clr_offmap_flag = cathay_expansionist_trend
				}
			}
		}

		if = {
			limit = { FROMFROM = { NOT = { has_offmap_flag = no_succession_news } } }
			any_player = {
				limit = {
					has_offmap_news_enabled = FROMFROM
					is_within_diplo_range = ROOT
				}
				narrative_event = { id = JD.50001 }
			}
		}

		FROMFROM = {
			clr_offmap_flag = no_policy_news
			clr_offmap_flag = no_succession_news
		}
	}
}
