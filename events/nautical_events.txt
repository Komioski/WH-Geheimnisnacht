#Hostile Admiral Spotted - Attacker
character_event = {
	id = nautical.1
	title = "nautical.1_title"
	desc = "nautical.1_desc"
	picture = GFX_evt_longships_at_sea_oldgods
	sound = bell
	is_triggered_only = yes

	option = {
		name = "nautical.1A"

		trigger = {
			war_with = event_target:nb_defending_admiral #The defending admiral
		}

		nb_ram_end_sounds_se = yes
		nb_assign_naval_outcome1_se = yes

		event_target:nb_defending_admiral = {
			save_event_target_as = nb_defending_admiral #The defending admiral
		}

		character_event = {
			id = nautical.2 #Begin Naval Battle - Attacker
		}

		ai_chance = {
			factor = 1

			modifier = {
				factor = 0

				martial < event_target:nb_defending_admiral #The defending admiral
			}
		}
	}

	option = {
		name = "nautical.1B"

		trigger = {
			NOT = {
				war_with = event_target:nb_defending_admiral #The defending admiral
			}
		}

		nb_ram_end_sounds_se = yes
		nb_assign_naval_outcome1_se = yes

		if = {
			limit = {
				NOR = {
					trait = viking
					trait = pirate
					trait = ravager
					trait = seaking
					trait = sea_queen
				}
			}
			random = {
				chance = 20

				if = {
					limit = {
						culture_group = norscan_group
					}
					add_trait = viking
				}
				else = {
					add_trait = pirate
				}
			}
		}

		reverse_opinion = {
			who = event_target:nb_defending_admiral #The defending admiral
			modifier = opinion_looted
		}

		event_target:nb_defending_admiral = {
			save_event_target_as = nb_defending_admiral #The defending admiral
		}

		character_event = {
			id = nautical.2 #Begin Naval Battle - Attacker
		}

		ai_chance = {
			factor = 1

			modifier = {
				factor = 0

				martial < event_target:nb_defending_admiral #The defending admiral
			}
		}
	}

	option = {
		name = "nautical.1C"

		event_target:nb_defending_admiral = {
			character_event = {
				id = nautical.4 #Hostile Admiral Spotted - Defender
			}
		}

		ai_chance = {
			factor = 1

			modifier = {
				factor = 0

				martial = event_target:nb_defending_admiral #The defending admiral
			}
		}
	}
}

character_event = {
	id = nautical.2
	title = "nautical.2_title"
	desc = "nautical.2_desc"
	picture = GFX_evt_longships_at_sea_oldgods
	is_triggered_only = yes
	hide_from = yes

	option = {
		name = "nautical.2A"

		trigger = {
			has_character_flag = nb_naval_victor
		}

		event_target:nb_defending_liege = {
			show_scope_change = no
			any_realm_province = {
				show_scope_change = no
				add_province_modifier = {
					modifier = nb_naval_supplies_em
					years = 10
					stacking = no
				}
			}
		}

		#victory_sounds_se = yes
		nb_naval_outcome1_se = yes
	}

	option = {
		name = "nautical.2B"

		trigger = {
			NOT = {
				has_character_flag = nb_naval_victor
			}
		}

		event_target:nb_attacking_liege = {
			show_scope_change = no
			any_realm_province = {
				show_scope_change = no
				add_province_modifier = {
					modifier = nb_naval_supplies_em
					years = 10
					stacking = no
				}
			}
		}

		}

#Ping for Admiral Spotted - Attacker
character_event = {
	id = nautical.100
	is_triggered_only = yes
	hide_window = yes
	in_command = yes

	trigger = {
		is_at_sea = yes
		is_sub_unit_leader = no
	}

	after = {
		if = {
			limit = {
				OR = {
					government = pirate_government
					government = freeboota_government
					government = malekith_government
					government = druchii_government
					culture_group = norscan_group
					trait = viking
					trait = pirate
					trait = ravager
					trait = seaking
					trait = sea_queen
				}
			}
			random_character = {
				limit = {
					in_command = yes
					rebel = no
					at_location = ROOT
					NOT = {
						same_realm = ROOT
					}
				}
				if = {
					limit = {
						top_liege = {
							any_current_enemy = {
								character = ROOT
							}
						}
					}
					top_liege = {
						save_event_target_as = nb_defending_liege #The defending liege
					}
				}
				else_if = {
					limit = {
						liege = {
							any_current_enemy = {
								character = ROOT
							}
						}
					}
					liege = {
						save_event_target_as = nb_defending_liege #The defending liege
					}
				}
				else = {
					save_event_target_as = nb_defending_liege #The defending liege
				}
				save_event_target_as = nb_defending_admiral #The defending admiral

				ROOT = {
					if = {
						limit = {
							top_liege = {
								any_current_enemy = {
									character = event_target:nb_defending_liege #The defending liege
								}
							}
						}
						top_liege = {
							save_event_target_as = nb_attacking_liege #The attacking liege
						}
					}
					else_if = {
						limit = {
							liege = {
								any_current_enemy = {
									character = event_target:nb_defending_liege #The defending liege
								}
							}
						}
						liege = {
							save_event_target_as = nb_attacking_liege #The attacking liege
						}
					}
					else = {
						save_event_target_as = nb_attacking_liege #The attacking liege
					}
					save_event_target_as = nb_attacking_admiral #The attacking admiral

					character_event = {
						id = nautical.1 #Hostile Admiral Spotted - Attacker
					}
				}
			}
		}
		else = {
			random_character = {
				limit = {
					in_command = yes
					rebel = no
					war_with = ROOT
					at_location = ROOT
				}
				if = {
					limit = {
						top_liege = {
							any_current_enemy = {
								character = ROOT
							}
						}
					}
					top_liege = {
						save_event_target_as = nb_defending_liege #The defending liege
					}
				}
				else_if = {
					limit = {
						liege = {
							any_current_enemy = {
								character = ROOT
							}
						}
					}
					liege = {
						save_event_target_as = nb_defending_liege #The defending liege
					}
				}
				else = {
					save_event_target_as = nb_defending_liege #The defending liege
				}
				save_event_target_as = nb_defending_admiral #The defending admiral

				ROOT = {
					if = {
						limit = {
							top_liege = {
								any_current_enemy = {
									character = event_target:nb_defending_liege #The defending liege
								}
							}
						}
						top_liege = {
							save_event_target_as = nb_attacking_liege #The attacking liege
						}
					}
					else_if = {
						limit = {
							liege = {
								any_current_enemy = {
									character = event_target:nb_defending_liege #The defending liege
								}
							}
						}
						liege = {
							save_event_target_as = nb_attacking_liege #The attacking liege
						}
					}
					else = {
						save_event_target_as = nb_attacking_liege #The attacking liege
					}
					save_event_target_as = nb_attacking_admiral #The attacking admiral

					character_event = {
						id = nautical.1 #Hostile Admiral Spotted - Attacker
					}
				}
			}
		}
	}
}
