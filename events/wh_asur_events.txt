character_event = {
	id = asur.4
	desc = EVTDESCasur4
	picture = GFX_evt_malekith
	sound = church_bells

	is_triggered_only = yes

	trigger = {
	}

	option = {
		name = EVTOPTAasur4
		prestige = -500
		add_trait = asuryan_stain
		add_trait = burnt_by_asuryan
		# Apply burnt by flames bloodline
		character_event = { id = aiclean.45 }
		if = {
			limit = { is_invincible_trigger = yes }
			antideath_effect = yes
			break = yes
		}
		death = { death_reason = death_suicide }
		hidden_tooltip = {
			#inform vassals
			any_vassal = {
				character_event = { id = asur.9 }
			}
			#already born children acquire stain
			any_child = {
				limit = { religion_group = ROOT }
				character_event = { id = asur.10 days = 2 }
			}
			#another Prince rather than the heir is granted the throne
			any_vassal = {
				limit = {
					tier = KING
					is_feudal = yes
					is_female = no		#don't scope the bloody Everqueen
					is_adult = yes
					trait = creature_elf
					NOT = { dynasty = ROOT }
					NOT = { trait = asuryan_stain }
				}
				character_event = { id = asur.11 days = 7 }
			}
		}
	}
}
