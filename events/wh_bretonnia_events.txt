character_event = {
	id = bretonnia.2
	hide_window = yes
	is_triggered_only = yes
	religion = grail
	trigger = {
		is_member_of_bretonnia_trigger = yes
	}
	immediate = {
		if = {
			limit = {
				society_rank == 1
			}
			add_trait = bretonnian_knight_errant
		}
		if = {
			limit = {
				society_rank == 2
			}
			add_character_modifier = {
				name = virtue_chivalry
				duration = -1
			}
			add_trait = bretonnian_knight
		}
		if = {
			limit = {
				society_rank == 3
			}
			add_trait = bretonnian_knight
			add_character_modifier = {
				name = virtue_chivalry
				duration = -1
			}
			add_character_modifier = {
				name = virtue_quest
				duration = -1
			}
			set_character_flag = virtue_quest
		}
		if = {
			limit = {
				society_rank == 4
			}
			add_character_modifier = {
				name = virtue_chivalry
				duration = -1
			}
			add_trait = grail_knight
		}
	}
}
