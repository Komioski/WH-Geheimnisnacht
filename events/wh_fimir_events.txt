namespace = fimir

#this event ensures that all children sired by a fimir are fimir.
character_event = {
	id = fimir.1
	hide_window = yes
	trigger = {
		OR = {
			mother_even_if_dead = { trait = creature_fimir }
			father_even_if_dead = { trait = creature_fimir }
		}
	}
	is_triggered_only = yes

	option = {
		remove_trait = creature_human
		remove_trait = creature_amazon
		add_trait = creature_fimir
		if = {
			limit = { NOT = { religion_group = chaos_gods_group } }
			religion = chaos
		}
		culture = fimir
		set_graphical_culture = fimir
		if = {
			limit = {
				ROOT = { is_female = yes }
				}
				set_gender = opposite
		}

	}
}

#Caste assignment event. This *should* also work for adult fimir who come to court and it will assign them a caste.
character_event = {
	id = fimir.2
	hide_window = yes

	trigger = {
		age = 2
		AND = {
			trait = creature_fimir
			NOR = {
				trait = fimir_shearl
				trait = fimir_fimm
				trait = fimir_noble
				trait = fimir_dirach
				trait = fimir_maergh
			}
		}
	}
	option = {
		random_list = {
			30 = { character_event = { id = fimir.20011 } }
			50 = { character_event = { id = fimir.20021 } }
			10 = { character_event = { id = fimir.20031 } }
			10 = { character_event = { id = fimir.20041 } }
		}
	}
}

#assigns caste to adult fimir. Doesn't need a notification so trait assignments are directly inside the random_list
character_event = {
	id = fimir.3
	hide_window = yes

	trigger = {
		AND = {
			age = 16
			trait = creature_fimir
			NOR = {
				trait = fimir_shearl
				trait = fimir_fimm
				trait = fimir_noble
				trait = fimir_dirach
				trait = fimir_maergh
			}
		}
	}
		option = {
		random_list = {
			30 = { add_trait = fimir_shearl	}
			50 = {
				mult_modifier = {
					factor = 0.0
					martial < 12
				}
				mult_modifier = {
					factor = 1.2
					martial = 12
				}
				mult_modifier = {
					factor = 1.5
					martial = 16
				}
			add_trait = fimir_fimm }
			10 = {
				mult_modifier = {
					factor = 0.0
					diplomacy < 12
				}
				mult_modifier = {
					factor = 1.2
					martial = 12
				}
				mult_modifier = {
					factor = 1.2
					diplomacy = 12
				}
				mult_modifier = {
					factor = 1.5
					diplomacy = 16
				}
			add_trait = fimir_noble }
			10 = {
				mult_modifier = {
					factor = 0.0
					intrigue < 12
				}
				mult_modifier = {
					factor = 1.2
					intrigue = 12
				}
				mult_modifier = {
					factor = 1.5
					intrigue = 16
				}
					mult_modifier = {
					factor = 1.5
					z_has_magic_power_true = yes
				}
			add_trait = fimir_dirach }
			fallback = { add_trait = shearl }
		}
	}
}

#backup event to assign creature trait
character_event = {
	id = fimir.4
	hide_window = yes

	trigger = {
		AND = {
			culture = fimir
			NOT = { trait = creature_fimir }
		}
	}

	immediate = {
		add_trait = creature_fimir
		if = {
			limit = {
				AND = {
					is_female = yes
					NOT = { trait = fimir_maergh }
				}
				set_gender = opposite
			}
		}
	}
}
