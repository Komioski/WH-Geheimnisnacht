# Wolf Hunt
character_event = {
	id = norsca.1
	title = WOLF_HUNT	# Wolf Hunt
	desc = EVTDESCNORSCA.1	# Word has reached you that wolves have begun terrorizing a local village, as their leader, it is your duty to stop them.
