#Adding maiden an scion traits for elves
character_event = {
	id = wh_racmarlog.2
	hide_window = yes

	ai = yes
	is_married = no

	mean_time_to_happen = {
		days = 5
	}

	trigger = {
		ai = yes
		OR = {
			graphical_culture = asurgfx
			graphical_culture = nagarythiangfx
			graphical_culture = caledoriangfx
			graphical_culture = asraigfx
			graphical_culture = eonirgfx
			graphical_culture = druchiigfx
		}
		NOR = {
			trait = elven_scion
			trait = elven_maiden
		}
		age = 12
		NOT = {
			age = 45
		}
		is_married = no
	}

	immediate = {
		if = {
			limit = {
				is_female = yes
			}
			add_trait = elven_maiden
		}
		if = {
			limit = {
				is_female = no
			}
			add_trait = elven_scion
		}
	}
}

#Beastmen concubine pruning
character_event = {
	id = wh_racmarlog.3
	hide_window = yes

	ai = yes

	mean_time_to_happen = {
		days = 5
	}

	trigger = {
		ai = yes

		}

			trait = creature_beastman
		}
	}
	
	immediate = {
		any_consort = {
			limit = {

			}
			remove_consort = ROOT
		}
	}
}
