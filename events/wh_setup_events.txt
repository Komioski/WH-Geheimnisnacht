					set_special_character_title = CORSAIR_PRINCE
				}

			}
			else_if = {

					tier = EMPEROR
				}
				if = {
					limit = { is_female = yes }
					set_special_character_title = CORSAIR_QUEEN
				}
				else = {
					limit = { is_female = no }
					set_special_character_title = CORSAIR_KING
				}
