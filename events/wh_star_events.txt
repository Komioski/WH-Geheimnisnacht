# Ping umbrella event for quickening journey
character_event = {
	id = stars.8
	hide_window = yes

	religion_group = old_world_group
	has_character_flag = quickening

	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				age >= 13
			}
			# Return home safe and sound
			character_event = {
				id = stars.7
				random = 25
			}
		}
		else = {
			random_list = {
				# Traveler nearby - potential food source if you handle this right... or you could starve
				20 = {
					character_event = {
						id = stars.10
						days = 5
						random = 5
					}
				}
				# Meet a Lost hunter - potential food source if you handle this right... or you could starve
				20 = {
					character_event = {
						id = stars.11
						days = 5
						random = 5
					}
				}
				# A Dieing Kid - potential food source if you handle this right... or you could starve
				20 = {
					character_event = {
						id = stars.12
						days = 5
						random = 5
					}
				}
				# Need for Food - forage, hunt, or cry about it
				20 = {
					modifier = {
						factor = 2
						OR = {
							trait = is_malnourished
							trait = feeble
						}
					}
					character_event = {
						id = stars.14
						days = 5
						random = 5
					}
				}
				# Being Wounded
				20 = {
					modifier = {
						factor = 2
						OR = {
							trait = is_malnourished
							trait = feeble
						}
					}
					# Being Wounded
					character_event = {
						id = stars.17
						days = 5
						random = 5
					}
				}
			}
		}
	}

	option = {
		name = OK
	}
}
